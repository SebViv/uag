 60      * @returns {string} Key string.

 62     uAgBasket.Basket.getKeyFromDate = function(date) {

 66     /**
 67      * @description Generate dates from storage keys.
 68      * @param {string} keyStr Key string.
 69      * @returns {string} Date object.
 70      */
 71     uAgBasket.Basket.getDateFromKey = function(keyStr) {
 72         var isoDateStr = keyStr.slice(uAgBasket.PREFIX_STR.length); // removing initial prefix
 73         return new Date(isoDateStr);
 74     };

303                 /**
304                  * @description Set current basket object using a JSON string input.
305                  * @param {string} str String of data in JSON format following the basket JSON Schema.
306                  * @returns {boolean} Success.
307                  */
308                 setCurrentBasketFromStr: function(str) {
309                     try {
310                         // checking if string is valid JSON
311                         var obj = JSON.parse(str);
312                         console.info('Info: input string is valid JSON');
313 
314                         // checking if resulting JSON object follows basket JSON schema
315                         var result = JSV_ENVIRONMENT_OBJ.validate(obj, JSON_SCHEMA_OBJ);
316                         if (result.errors.length === 0) { // success JSON schema
317                             console.info('Info: input string follows JSON schema for basket data');
318                             currentBasketObj = uAgBasket.Basket.decorate(obj);
319 
320                             // sanity check with timestamp
321                             var date = new Date(currentBasketObj.timestamp);
322                             currentBasketObj.timestamp = date.toISOString(); // converting to full ISO8601 format (with ms)
323                             return true;
324                         } else { // failure JSON schema
325                             var errorArr = result.errors;
326                             console.error('Error: file object does not follow JSON schema for basket data\n' +
327                                           'Error: > uri: ' + errorArr[0].uri + '\n' +
328                                           'Error: > message: ' + errorArr[0].message);
329                             return false;
330                         }
331                     } catch (e) { // failure JSON
332                         console.error(e.message);
333                         return false;
334                     }
335                 },

337                 /**
338                  * @description Set current basket object using a Date object input, as unique timestamp.
339                  * @param {date} date Date object defining basket timestamp.
340                  * @returns {boolean} Success.
341                  */
342                 setCurrentBasketFromDate: function(date) {
343                     var key = uAgBasket.Basket.getKeyFromDate(date);
344                     var basketStr = window.localStorage.getItem(key);
345                     if (basketStr !== null) {
346                         // sanity check with JSON parse
347                         try {
348                             var obj = JSON.parse(basketStr);
349                             currentBasketObj = uAgBasket.Basket.decorate(obj);
350                             return true;
351                         } catch (e) {
352                             console.error(e.message);
353                             return false;
354                         }
355                     } else {
356                         console.info('Info: creating new basket');
357                         currentBasketObj = new uAgBasket.Basket(date);
358                         return true;
359                     }
360                 },

365                  * @returns {boolean} Success.
366                  */
367                 setCurrentProduct: function(nbr) {
368                     if (currentBasketObj !== null) {
369                         var productsArr = currentBasketObj.products;
370                         if (nbr >= 0 && nbr < productsArr.length) { // take the empty array case into account
371                             currentProductObj = productsArr[nbr];
372                             return true;
373                         } else {
374                             console.error('Error: no current basket products available');
375                             return false;
376                         }
377                     } else {
378                         console.error('Error: no current basket available');
379                         return false;
380                     }
381                 },

385                  * @returns {boolean} Success.
386                  */
387                 storeCurrentBasket: function() {
388                     // sanity check with current basket
389                     if (currentBasketObj !== null) {
390 
391                         // sanity check with JSON schema
392                         var result = JSV_ENVIRONMENT_OBJ.validate(currentBasketObj, JSON_SCHEMA_OBJ);
393                         if (result.errors.length === 0) {
394 
395                             // sanity check with JSON stringify
396                             try {
397                                 var basketStr = JSON.stringify(currentBasketObj);
398 
399                                 // checking if key already exists in local storage
400                                 var key = currentBasketObj.getKey();
401                                 var value = window.localStorage.getItem(key);
402                                 if (value === null) {
403                                     console.info('Info: adding new key-value pair to local storage');
404                                     storedBasketKeysArr.push(key);
405                                 }
406                                 window.localStorage.setItem(key, basketStr);
407                                 return true;
408                             } catch (e) { // failure JSON
409                                 console.error(e.message);
410                                 return false;
411                             }
412                         } else { // failure JSON schema
413                             var errorArr = result.errors;
414                             console.error('Error: file object does not follow JSON schema for basket data\n' +
415                                           'Error: > uri: ' + errorArr[0].uri + '\n' +
416                                           'Error: > message: ' + errorArr[0].message);
417                             return false;
418                         }
419                     } else { // failure current basket
420                         console.error('Error: no basket available');
421                         return false;
422                     }
