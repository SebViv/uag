 31                 /** @ignore */
 32                 function pad(n) {
 33                     return (n < 10 ? '0' + n : n);
 34                 }
 35 
 36                 return d.getUTCFullYear() + '-' +
 37                     pad(d.getUTCMonth()+1) + '-' +
 38                     pad(d.getUTCDate()) + 'T' +
 39                     pad(d.getUTCHours()) + ':' +
 40                     pad(d.getUTCMinutes()) + ':' +
 41                     pad(d.getUTCSeconds()) + 'Z';
 42             }
