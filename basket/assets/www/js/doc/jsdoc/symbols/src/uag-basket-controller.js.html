 10 /** @namespace uAg project */
 11 var uag = (function (parent, $, window, document, FileExplorer) {
 12     'use strict'; // strict JS
 13 
 14     var uAgBasket = parent.basket = parent.basket || {};
 15     /**
 16      * @class
 17      * @exports uAgBasket.controller as uag.basket.controller
 18      * @description Controller of the Basket app
 19      */
 20     uAgBasket.controller = (function () {
 21         var isReadyCordova = false;
 22         var isReadyJQueryMobile = false;
 23         var fileExplorer = null;
 24         try {
 25             fileExplorer = new FileExplorer($('#open-back-btn'));
 26         } catch (e) {
 27             console.error(e.message);
 28         }
 29 
 30         /** @lends uag.basket.controller */
 31         return {
 32             onLoad: function (event) {
 33                 document.addEventListener('deviceready', this.onDeviceReady, false);
 34                 console.info('Info: deviceready event registered');
 35             },
 36             onDeviceReady: function (event) {
 37                 isReadyCordova = true;
 38                 document.addEventListener('pause', this.onPause, false);
 39                 console.info('Info: pause event registered');
 40                 try {
 41                     fileExplorer.initAfterCordova();
 42                 } catch (e) {
 43                     console.error(e.message);
 44                 }
 45             },
 46             onPause: function (event) {
 47                 // TODO
 48             },
 49             pageInit: function (event) {
 50                 isReadyJQueryMobile = true;
 51             },
 52             init: function () {
 53                 $(document).bind('pageinit', this.pageInit);
 54                 console.info('Info: pageinit event registered');
 55             },
 56             isReady: function () {
 57                 return (isReadyCordova && isReadyJQueryMobile);
 58             },
 59             checkState: function () {
 60                 // TMP
 61                 if (this.isReady()) {
 62                     console.log('Debug: uag.basket.controller ready');
 63                 } else {
 64                     console.log('Debug: uag.basket.controller NOT ready');
 65                 }
 66             }
 67         };
 68     }());
 69     Object.freeze(uAgBasket.controller); // final
 70 
 71     return parent;
 72 }(uag || {}, jQuery, this, this.document, uag.utils.FileExplorer));
