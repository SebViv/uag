 10 /** @namespace uAg project */
 11 var uag = (function(parent, $, window, document, undefined) {
 12     'use strict';
 13     // namespace declarations
 14     var uAgUtils = parent.utils = parent.utils || {};
 15     var uAgBasket = parent.basket = parent.basket || {};
 16 
 17     /**
 18      * @class
 19      * @exports uAgBasket.Controller as uag.basket.Controller
 20      * @description Controller of the Basket app (singleton pattern)
 21      */
 22     uAgBasket.Controller = (function() {
 23         var instance = null;
 24 
 25         /** @ignore */
 26         function init() {
 27             // view objects
 28             var platformSpan = null;
 29             var versionSpan = null;
 30             var uuidSpan = null;
 31             var pageDiv = null;
 32             var gridDiv = null;
 33 
 34             // other vars
 35             var isJQueryMobileReady = false;
 36             var isCordovaReady = false;
 37             var fileExplorer = uAgUtils.makeFileExplorer();
 38             var fileObj = null;
 39 
 40             /** @ignore */
 41             function onFileExplorerCheck(fileStr) {
 42                 try {
 43                     fileObj = JSON.parse(fileStr);
 44                     console.info('Info: file object is valid JSON');
 45                 } catch (e) {
 46                     console.error(e.message);
 47                     return false;
 48                 }
 49                 var env = JSV.createEnvironment("json-schema-draft-03"); // current default draft version
 50                 var result = env.validate(fileObj, uAgBasket.model.JSON_SCHEMA);
 51                 if (result.errors.length === 0) { // success
 52                     console.info('Info: file object follows JSON schema for basket data');
 53                     return true;
 54                 } else { // failure
 55                     var errorArr = result.errors;
 56                     console.error('Error: file object does not follow JSON schema for basket data\n' +
 57                                   'Error: > uri: ' + errorArr[0].uri + '\n' +
 58                                   'Error: > message: ' + errorArr[0].message);
 59                     return false;
 60                 }
 61             }
 62 
 63             /** @ignore */
 64             function onFileExplorerClose(event) {
 65                 if (pageDiv !== null) {
 66                     pageDiv.dialog('close');
 67                 } else {
 68                     console.error('Error: explorer view objects are not set');
 69                 }
 70             }
 71 
 72             /**
 73              * @public
 74              * @lends uag.basket.Controller
 75              */
 76             return {
 77                 /** @description Function callback invoked when jQuery Mobile starts. */
 78                 onMobileInit: function(event) {
 79                     console.info('Info: mobileinit event fired');
 80                     isJQueryMobileReady = true;
 81                     // add page loading symbol in jQuery Mobile 1.2
 82                     // TODO
 83                 },
 84 
 85                 /** @description Function callback invoked when a Cordova application is ready. */
 86                 onDeviceReady: function(event) {
 87                     console.info('Info: deviceready event fired');
 88                     isCordovaReady = true;
 89                     platformSpan.text(window.device.platform);
 90                     versionSpan.text(window.device.version);
 91                     uuidSpan.text(window.device.uuid);
 92                 },
 93 
 94                 /** @description Function callback invoked when a new page is loaded and created by jQuery Mobile. */
 95                 onPageInit: function(event) {
 96                     console.info('Info: pageinit event fired');
 97                 },
 98 
 99                 /** @description Function callback invoked when a Cordova application is put into the background. */
100                 onPause: function(event) {
101                     console.info('Info: pause event fired');
102                 },
103 
104                 /** @description Function callback invoked when a Cordova application is retrieved from the background. */
105                 onResume: function(event) {
106                     console.info('Info: resume event fired');
107                 },
108 
109                 /** @description TODO */
110                 options: function(event) {
111                 },
112 
113                 /** @description TODO */
114                 newBasket: function(event) {
115                 },
116 
117                 /** @description TODO */
118                 openBasket: function(event) {
119                     if (isCordovaReady) {
120                         console.info('Info: uag.basket.controller ready');
121                         try {
122                             fileExplorer.run(gridDiv,
123                                              onFileExplorerCheck,
124                                              onFileExplorerClose);
125                         } catch (e) {
126                             console.error(e.message);
127                         }
128                     } else {
129                         console.error('Error: uag.basket.controller NOT READY');
130                     }
131                 },
132 
133                 /** @description TODO */
134                 saveBasket: function(event) {
135                 },
136 
137                 /** @description TODO */
138                 captureTag: function(event) {
139                 },
140 
141                 /** @description TODO */
142                 addNewBasketItem: function(event) {
143                 },
144 
145                 /** @description TODO */
146                 loadBasketFile: function(event) {
147                 },
148 
149                 /**
150                  * @param {object} $platformSpan JQuery Mobile object containing the DOM reference to a span.
151                  * @param {object} $versionSpan JQuery Mobile object containing the DOM reference to a span.
152                  * @param {object} $uuidSpan JQuery Mobile object containing the DOM reference to a span.
153                  * @description Set the jQuery Mobile objects of the about dialog view.
154                  */
155                 setAboutView: function($platformSpan, $versionSpan, $uuidSpan) {
156                     if ($platformSpan instanceof jQuery && $platformSpan.is('span') &&
157                         $versionSpan instanceof jQuery && $versionSpan.is('span') &&
158                         $uuidSpan instanceof jQuery && $uuidSpan.is('span')) {
159                         platformSpan = $platformSpan;
160                         versionSpan = $versionSpan;
161                         uuidSpan = $uuidSpan;
162                     } else {
163                         console.error('Error: wrong about view info');
164                     }
165                 },
166 
167                 /**
168                  * @param {object} $pageDiv JQuery Mobile object containing the DOM reference to the div with the page role.
169                  * @param {object} $gridDiv JQuery Mobile object containing the DOM reference to the div with the grid class.
170                  * @description Set the jQuery Mobile objects of the file explorer view.
171                  */
172                 setExplorerView: function($pageDiv, $gridDiv) {
173                     // how to make sure the page div has been opened as a dialog?
174                     // TODO
175                     if ($pageDiv instanceof jQuery &&
176                         $pageDiv.is('div[data-role="page"]') &&
177                         $gridDiv instanceof jQuery &&
178                         $gridDiv.is('div[class="ui-grid-b"]')) {
179                         pageDiv = $pageDiv;
180                         gridDiv = $gridDiv;
181                     } else {
182                         console.error('Error: wrong about view info');
183                     }
184                 },
185             }; // return {}
186         }; // function init()
187 
188         /**
189          * @public
190          * @lends uag.basket.Controller
191          */
192         return {
193             /**
194              * @returns {object} Controller singleton instance.
195              * @description Get the Singleton instance if one exists or create one if it doesn't.
196              */
197             getInstance: function() {
198                 if (instance === null) {
199                     instance = init();
200                     Object.freeze(instance);
201                 }
202                 return instance;
203             },
204         }; // return {}
205     }()); // immediately-invoked function expression (IIFE)
206 
207     return parent;
208 }(uag || {}, jQuery, this, this.document));
