  5 /**
  6  * @fileOverview uAg Basket View
  7  * @author <a href="http://www.davidbourguignon.net">David Bourguignon</a>
  8  * @version 2012-09-04
  9  */
 10 /** @namespace uAg project */
 11 var uag = (function(parent, $, window, document, undefined) {
 12     'use strict';
 13     // namespace declarations
 14     var uAgUtils = parent.utils = parent.utils || {};
 15     var uAgBasket = parent.basket = parent.basket || {};
 16 
 17     /**
 18      * @class
 19      * @exports uAgBasket.View as uag.basket.View
 20      * @description View of the Basket app (singleton pattern)
 21      */
 22     uAgBasket.View = (function() {
 23         var instance = null;
 24 
 25         /** @ignore */
 26         function init() {
 27             var importFileExplorer = uAgUtils.makeFileExplorer();
 28 
 29             /** @ignore */
 30             function onHomePageInit(event) {
 31                 var controller = uAgBasket.Controller.getInstance();
 32                 $('#home-new-btn').on('click', controller.onNewBasketClick);
 33                 $('#home-open-btn').on('click', controller.onOpenBasketClick);
 34                 $('#home-import-btn').on('click', controller.onImportBasketClick);
 35             }
 36 
 37             /** @ignore */
 38             function onInfoPageInit(event) {
 39                 $('#device-platform').text(window.device.platform);
 40                 $('#device-version').text(window.device.version);
 41                 $('#device-uuid').text(window.device.uuid);
 42             }
 43 
 44             /** @ignore */
 45             function onEditPageInit(event) {
 46                 var controller = uAgBasket.Controller.getInstance();
 47                 $('#edit-save-btn').on('click', controller.onSaveBasketClick);
 48                 $('#edit-tag-btn').on('click', controller.onCaptureTagClick);
 49                 $('#edit-add-btn').on('click', controller.onAddBasketItemClick);
 50             }
 51 
 52             /** @ignore */
 53             function onOpenPageChange() {
 54                 var model = uAgBasket.Model.getInstance();
 55                 var dates = model.getStoredBasketDates();
 56                 var $ul = $('#open-basket-explorer-ul');
 57                 $ul.empty();
 58                 for (var i = 0, len = dates.length; i < len; i++) {
 59                     $ul.append('<li><a href="#edit">' + dates[i] + '</a></li>'); // list view syntax
 60                 }
 61                 var controller = uAgBasket.Controller.getInstance();
 62                 $ul.find('a').on('click', controller.onBasketExplorerClick);
 63             }
 64 
 65             /** @ignore */
 66             function onEditPageChange() {
 67                 var model = uAgBasket.Model.getInstance();
 68                 var basketObj = model.getCurrentBasket();
 69                 for (var key in basketObj) {
 70                     switch (key) {
 71                         case 'distribDate':
 72                             $('#edit-distrib-date').val(basketObj[key]); //////////////// PUT EVERYTHING IN CONST VARS!!!!!!!!!!!!!!!!!!!
 73                             break;
 74                         default:
 75                             //console.error('Error: unknown property in basket object');
 76                     }
 77                 }
 78             }
 79 
 80             /** @ignore */
 81             function pageDataToId(pageData) {
 82                 var pageId = '';
 83                 if (typeof pageData === 'object') {
 84                     pageId = pageData.attr('id');
 85                 } else if (typeof pageData === 'string') {
 86                     var urlObj = $.mobile.path.parseUrl(pageData);
 87                     pageId = urlObj.hash;
 88                     pageId = pageId.slice(1); // removing initial #
 89                 } else {
 90                     console.error('Error: unknown type for data.toPage');
 91                 }
 92                 return pageId;
 93             }
 94 
 95             /**
 96              * @public
 97              * @lends uag.basket.View
 98              */
 99             return {
100                 /**
101                  * @description TODO
102                  */
103                 onPageInit: function(event) {
104                     var pageId = $(event.target).attr('id');
105                     console.info('Info: pageinit event fired for ' + pageId);
106                     switch (pageId) {
107                         case 'home':
108                             onHomePageInit();
109                             break;
110                         case 'info':
111                             onInfoPageInit();
112                             break;
113                         case 'open':
114                         case 'import':
115                             // TODO
116                             break;
117                         case 'edit':
118                             onEditPageInit();
119                             break;
120                         default:
121                             console.error('Error: unknown page id');
122                     }
123                 },
124 
125                 /**
126                  * @description TODO
127                  */
128                 onPageChange: function(event, data) {
129                     var pageId = pageDataToId(data.toPage);
130                     console.info('Info: pagechange event fired for ' + pageId);
131                     switch (pageId) {
132                         case 'home':
133                         case 'info':
134                             break;
135                         case 'open':
136                             onOpenPageChange();
137                             break;
138                         case 'import':
139                             break;
140                         case 'edit':
141                             onEditPageChange();
142                             break;
143                         default:
144                             console.error('Error: unknown page id');
145                     }
146                 },
147 
148                 /**
149                  * @description TODO
150                  */
151                 onPageChangeFailed: function(event, data) {
152                     console.error('Error: pagechangefailed event fired for '  + pageDataToId(data.toPage));
153                 },
154 
155                 /**
156                  * @description TODO
157                  */
158                 changeToOpenPage: function() {
159                     // TODO
160                 },
161 
162                 /**
163                  * @description TODO
164                  */
165                 changeToImportPage: function() {
166                     var controller = uAgBasket.Controller.getInstance();
167                     try {
168                         importFileExplorer.run($('#import-file-explorer-div'),
169                                                controller.onFileExplorerCheck);
170                     } catch (e) {
171                         console.error(e.message);
172                     }
173                 },
174 
175                 /**
176                  * @description TODO
177                  */
178                 changeToEditPage: function() {
179                     $.mobile.changePage($('#edit'), {transition:'fade'}); //////////////// PUT EVERYTHING IN CONST VARS!!!!!!!!!!!!!!!!!!!
180                 },
181 
182                 /**
183                  * @description TODO
184                  */
185                 resetEditPage: function() {
186                     $('#edit-distrib-date').val(''); //////////////// PUT EVERYTHING IN CONST VARS!!!!!!!!!!!!!!!!!!!
187                 },
188             }; // return
189         } // private function init()
190 
191         /**
192          * @public
193          * @lends uag.basket.View
194          */
195         return {
196             /**
197              * @returns {object} View singleton instance.
198              * @description Get the Singleton instance if one exists or create one if it doesn't.
199              */
200             getInstance: function() {
201                 if (instance === null) {
202                     instance = init();
203                     Object.freeze(instance);
204                 }
205                 return instance;
206             },
207         }; // return
208     }()); // immediately-invoked function expression (IIFE)
209 
210     return parent;
211 }(uag || {}, jQuery, this, this.document));
