 10 /** @namespace uAg project */
 11 var uag = (function(parent, $, window, document, undefined) {
 12     'use strict';
 13     // namespace declarations
 14     var uAgUtils = parent.utils = parent.utils || {};
 15     var uAgBasket = parent.basket = parent.basket || {};
 16 
 17     /**
 18      * @class
 19      * @exports uAgBasket.View as uag.basket.View
 20      * @description View of the Basket app (singleton pattern)
 21      */
 22     uAgBasket.View = (function() {
 23         var instance = null;
 24 
 25         /** @ignore */
 26         function init() {
 27             var importFileExplorer = uAgUtils.makeFileExplorer();
 28 
 29             /** @ignore */
 30             function bindController() {
 31                 var controller = uAgBasket.Controller.getInstance();
 32 
 33                 // jQuery Mobile bindings
 34                 $(document).on('mobileinit', controller.onMobileInit); // never called: too late? TODO
 35                 $(document).on('pageinit', controller.onPageInit);
 36 
 37                 // Cordova bindings
 38                 $(document).on('deviceready', controller.onDeviceReady);
 39                 $(document).on('pause', controller.onPause);
 40                 $(document).on("resume", controller.onResume);
 41 
 42                 // other bindings, when DOM is ready
 43                 $(document).ready(function() {
 44                     $('#home-new-btn').on('click', controller.onNewBasketClick);
 45                     $('#home-open-btn').on('click', controller.onOpenBasketClick);
 46                     $('#home-import-btn').on('click', controller.onImportBasketClick);
 47                     $('#new-save-btn').on('click', controller.onSaveBasketClick);
 48                     $('#new-tag-btn').on('click', controller.onCaptureTagClick);
 49                     $('#new-add-btn').on('click', controller.onAddBasketItemClick);
 50                     // managing open save/tag btn differently from new?
 51                     // TODO
 52                 });
 53             }
 54 
 55             // initial setup
 56             bindController();
 57 
 58             /**
 59              * @public
 60              * @lends uag.basket.View
 61              */
 62             return {
 63                 /**
 64                  * @description TODO
 65                  */
 66                 setDeviceInfo: function() {
 67                     // how to make sure this is called after onDeviceReady has fired?
 68                     // TODO
 69                     $('#device-platform').text(window.device.platform);
 70                     $('#device-version').text(window.device.version);
 71                     $('#device-uuid').text(window.device.uuid);
 72                 },
 73                 /**
 74                  * @description TODO
 75                  */
 76                 openImportFileExplorer: function() {
 77                     var controller = uAgBasket.Controller.getInstance();
 78                     importFileExplorer.run($('#import-file-explorer-div'),
 79                                            controller.onFileExplorerCheck);
 80                 },
 81                 /**
 82                  * @description TODO
 83                  */
 84                 switchToEditPage: function() {
 85                     $.mobile.changePage($('#edit'), {transition:'fade'});
 86                     var model = uAgBasket.Model.getInstance();
 87                     //request info from model to populate page
 88                 },
 89             }; // return
 90         } // private function init()
 91 
 92         /**
 93          * @public
 94          * @lends uag.basket.View
 95          */
 96         return {
 97             /**
 98              * @returns {object} View singleton instance.
 99              * @description Get the Singleton instance if one exists or create one if it doesn't.
100              */
101             getInstance: function() {
102                 if (instance === null) {
103                     instance = init();
104                     Object.freeze(instance);
105                 }
106                 return instance;
107             },
108         }; // return
109     }()); // immediately-invoked function expression (IIFE)
110 
111     return parent;
112 }(uag || {}, jQuery, this, this.document));
