 48             /** @ignore */
 49             function onOpenPageChange() {
 50                 var model = uAgBasket.Model.getInstance();
 51                 var dates = model.getStoredBasketDates();
 52                 var $openTitle = $('#open-basket-explorer-title'); // TODO CONST VAR
 53                 var $basketList = $('#open-basket-explorer-listview'); // TODO CONST VAR
 54                 var datesNbr = dates.length;
 55                 if (datesNbr === 0) {
 56                     $openTitle.text('No baskets in store!');
 57                 } else {
 58                     for (var i = 0; i < datesNbr; i++) {
 59                         var date = dates[i];
 60                         $basketList.append('<li><a id="' +
 61                                            date.toISOString() +
 62                                            '" href="#edit">' +
 63                                            date.toLocaleDateString() +
 64                                            '</a></li>');
 65                     }
 66                     $basketList.listview('refresh');
 67                     var controller = uAgBasket.Controller.getInstance();
 68                     $basketList.find('a').on('click', controller.onBasketListClick);
 69                 }
 70             }

 86             /** @ignore */
 87             function onEditPageChange() {
 88                 var model = uAgBasket.Model.getInstance();
 89                 var basketObj = model.getCurrentBasket();
 90                 if (basketObj !== null) {
 91                     var $editTitle = $('#edit-basket-title'); // TODO CONST VAR
 92                     var $productList = $('#edit-basket-listview'); // TODO CONST VAR
 93                     var productsLen = basketObj.products.length;
 94                     if (productsLen === 0) {
 95                         $editTitle.text('No products in basket!');
 96                     } else {
 97                         var basketWeight = 0;
 98                         for (var i = 0; i < productsLen; i++) {
 99                             var productObj = basketObj.products[i];
100                             var isIn = productObj.isIn;
101                             var name = productObj.name;
102                             var weight = productObj.weight;
103                             var id = uAgBasket.Product.PREFIX_STR + i;
104                             $productList.append('<li' +
105                                                 (isIn ? ' data-theme="b"' : '') +
106                                                 '><a id="' + id +
107                                                 '" href="#product">' +
108                                                 name + ', ' + weight +
109                                                 'kg</a></li>');
110                             basketWeight += weight;
111                         }
112                         $editTitle.text(productsLen + ' product' +
113                                         (productsLen > 1 ? 's, ' : ', ') +
114                                         basketWeight + 'kg');
115                         $productList.listview('refresh');
116                         var controller = uAgBasket.Controller.getInstance();
117                         $productList.find('a').on('click', controller.onProductListClick);
118                     }
119                 } else {
120                     console.error('Error: no current basket available');
121                 }
122             }

145             /** @ignore */
146             function onProductPageChange() {
147                 var model = uAgBasket.Model.getInstance();
148                 var productObj = model.getCurrentProduct();
149                 if (productObj !== null) {
150                     // change fields
151                     $('#product-isIn').val(productObj.isIn.toString());
152                     $('#product-name').val(productObj.name);
153                     $('#product-producerName').val(productObj.producerName);
154                     $('#product-weight').val(productObj.weight);
155                     // refresh widgets
156                     $('#product-isIn').slider('refresh');
157                     $('#product-weight').slider('refresh');
158                     // change buttons
159                     if (productObj.tag.text === '') {
160                         $('#product-tag-btn').text('Scan tag');
161                     } else {
162                         $('#product-tag-btn').text('Open tag');
163                     }
164                 } else {
165                     console.error('Error: no current product available');

174             /** @ignore */
175             function onTagPageChange() {
176                 var model = uAgBasket.Model.getInstance();
177                 var productObj = model.getCurrentProduct();
178                 if (productObj !== null && productObj.tag.text !== '') {
179                     var $tagDiv = $('#tag-div');
180                     var text = productObj.tag.text;
181                     if ($.mobile.path.isAbsoluteUrl(text)) {
182                         // display URL
183                         $tagDiv.append('<h3>URL</h3>');
184                         $tagDiv.append('<p><a href="' + text +
185                                        '" data-rel="external">' + text +
186                                        '</a></p>');
187                     } else if (uAgUtils.isRFC822ValidEmail(text)) {
188                         //display email address
189                         // TODO TEST
190                         $tagDiv.append('<h3>Email address</h3>');
191                         $tagDiv.append('<p><a href="mailto:' + text +
192                                        '?Subject=Your%20product%20barcode">' +
193                                        text + '</a></p>');
194                     } else if (uAgUtils.isValidPhoneNumber(text)) {
195                         //display email
196                         // TODO TEST
197                         $tagDiv.append('<h3>Phone number</h3>');
198                         $tagDiv.append('<p><a href="tel:' + text +
199                                        '">' + text + '</a></p>');
200                         /**
201                          * See http://tools.ietf.org/html/rfc5341 about the 'tel' Uniform Resource Identifier.
202                          * On desktop computers the non-standard <a href="callto: is used by Skype only.
203                          */
204                     } else {
205                         // display plain text
206                         // TODO TEST
207                         $tagDiv.append('<h3>Text</h3>');
208                         $tagDiv.append('<p>' + text + '</p>');
209                     }
210                 } else {
211                     console.error('Error: no tag info available');
212                 }

215             /** @ignore */
216             function pageDataToId(pageData) {
217                 var pageId = '';
218                 if (typeof pageData === 'object') {
219                     pageId = pageData.attr('id');
220                 } else if (typeof pageData === 'string') {
221                     var urlObj = $.mobile.path.parseUrl(pageData);
222                     pageId = urlObj.hash;
223                     pageId = pageId.slice(1); // removing initial #
224                 } else {
225                     console.error('Error: unknown type for data.toPage');
226                 }
227                 return pageId;
228             }
