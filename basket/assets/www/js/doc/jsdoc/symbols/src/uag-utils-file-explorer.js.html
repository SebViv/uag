  5 /**
  6  * @fileOverview uAg Utils File Explorer
  7  * @author <a href="http://www.davidbourguignon.net">David Bourguignon</a>
  8  * @version 2012-08-31
  9  */
 10 /** @namespace uAg project */
 11 var uag = (function(parent, $, window, document, undefined) {
 12     'use strict';
 13     // namespace declaration
 14     var uAgUtils = parent.utils = parent.utils || {};
 15 
 16     /**
 17      * @class
 18      * @returns File explorer object
 19      * @exports uAgUtils.makeFileExplorer as uag.utils.makeFileExplorer
 20      * @description File explorer is a view component containing a small model and controller inside.
 21      */
 22     uAgUtils.makeFileExplorer = function() {
 23         /**
 24          * @private
 25          */
 26         // view objects
 27         var gridDiv = null;
 28         var folderDiv = null;
 29         var fileDiv = null;
 30         var backDiv = null;
 31 
 32         // file API objects
 33         var rootDir = null;
 34         var currentDir = null;
 35         var parentDir = null;
 36         var activeEntry = null;
 37 
 38         // callbacks from caller
 39         var onFileCheck = null;
 40 
 41         // other vars
 42         var directoryEntries = [];
 43         var fileEntries = [];
 44 
 45         // callback functions
 46         /** @ignore */
 47         function onFileError(fileError) {
 48             throw new Error('Error: failed with code ' + fileError.code);
 49         }
 50 
 51         /** @ignore */
 52         function onRequestFileSystemSuccess(fileSystem) {
 53             rootDir = fileSystem.root;
 54             console.info('Info: root directory name is ' + rootDir.name);
 55             showDirectory(rootDir);
 56         }
 57 
 58         /** @ignore */
 59         function onGetParentSuccess(parent) {
 60             parentDir = parent;
 61             console.info('Info: parent directory name is ' + parentDir.name);
 62         }
 63 
 64         /** @ignore */
 65         function onReadEntriesSuccess(entries) {
 66             directoryEntries.length = 0; // emptying arrays
 67             fileEntries.length = 0;
 68             var i, len; // vars for loops
 69             for (i = 0, len = entries.length; i < len; i++) {
 70                 var entry = entries[i];
 71                 if (entry.isDirectory && entry.name[0] !== '.') { // removing hidden files
 72                     directoryEntries.push(entry);
 73                 } else if (entry.isFile && entry.name[0] !== '.') {
 74                     fileEntries.push(entry);
 75                 }
 76             }
 77             var sortedEntries = directoryEntries.concat(fileEntries);
 78             gridDiv.empty();
 79             gridDiv.append('<h4> Folder ' + currentDir.name + '</h4>');
 80             var uiBlockLetters = ['a','b','c'];
 81             for (i = 0, len = sortedEntries.length; i < len; i++) {
 82                 var blockLetter = uiBlockLetters[i%3]; // length uiBlockLetters = 3
 83                 var sortedEntry = sortedEntries[i];
 84                 if (sortedEntry.isDirectory) {
 85                     gridDiv.append('<div class="ui-block-' +
 86                                    blockLetter +
 87                                    '"><div class="folder"><p>' +
 88                                    sortedEntry.name +
 89                                    '</p></div></div>');
 90                 } else if (sortedEntry.isFile) {
 91                     gridDiv.append('<div class="ui-block-' +
 92                                    blockLetter +
 93                                    '"><div class="file"><p>' +
 94                                    sortedEntry.name +
 95                                    '</p></div></div>');
 96                 }
 97             }
 98             folderDiv = $('.folder');
 99             fileDiv = $('.file');
100             folderDiv.on('click', onFolderDivClick);
101             fileDiv.on('click', onFileDivClick);
102             if (currentDir !== null && currentDir.name !== rootDir.name) {
103                 gridDiv.append('<div class="ui-block-a"><div class="folder back"><p>..</p></div></div>');
104                 backDiv = $('.folder.back');
105                 backDiv.on('click', onBackDivClick);
106             }
107         }
108 
109         /** @ignore */
110         function onGetDirectorySuccess(directoryEntry) {
111             activeEntry = directoryEntry;
112             showDirectory(activeEntry);
113         }
114 
115         /** @ignore */
116         function onFolderDivClick(event) {
117             var name = $(event.target).text();
118             if (currentDir !== null) {
119                 currentDir.getDirectory(name, {create:false},
120                                         onGetDirectorySuccess,
121                                         onFileError);
122             } else {
123                 throw new Error('Error: current directory does not exist');
124             }
125 
126         }
127 
128         /** @ignore */
129         function onGetFileSuccess(fileEntry) {
130             activeEntry = fileEntry;
131             readFile(activeEntry);
132         }
133 
134         /** @ignore */
135         function onFileDivClick(event) {
136             var name = $(event.target).text();
137             if (currentDir !== null) {
138                 currentDir.getFile(name, {create:false},
139                                    onGetFileSuccess,
140                                    onFileError);
141             } else {
142                 throw new Error('Error: current directory does not exist');
143             }
144         }
145 
146         /** @ignore */
147         function onBackDivClick(event) {
148             if (parentDir !== null) {
149                 showDirectory(parentDir);
150             }
151         }
152 
153         /** @ignore */
154         function onFileReaderLoadEnd(event) {
155             var fileReadStr = event.target.result;
156             console.info('Info: > file content is');
157             console.info(fileReadStr);
158             var isFileChecked = onFileCheck(fileReadStr);
159             if (!isFileChecked) {
160                 // use jQuery Mobile 1.2.0 popup
161                 // TODO
162                 console.warning('Warning: file is not checked');
163             }
164         }
165 
166         /** @ignore */
167         function onFileSuccess(file) {
168             console.info('Info: file details\n' +
169                          'Info: > file name is ' + file.name + '\n' +
170                          'Info: > file type is ' + file.type + '\n' +
171                          'Info: > file date is ' + new Date(file.lastModifiedDate) + '\n' +
172                          'Info: > file size is ' + file.size + ' bytes');
173             var reader = new FileReader();
174             reader.onloadend = onFileReaderLoadEnd;
175             reader.readAsText(file);
176         }
177 
178         // other functions
179         /** @ignore */
180         function showDirectory(directoryEntry) {
181             if (directoryEntry.isDirectory) {
182                 currentDir = directoryEntry;
183                 console.info('Info: current directory name is ' + currentDir.name);
184                 directoryEntry.getParent(onGetParentSuccess, onFileError);
185             } else {
186                 throw new TypeError('Error: expecting directory object');
187             }
188             var directoryReader = directoryEntry.createReader();
189             directoryReader.readEntries(onReadEntriesSuccess,
190                                         onFileError);
191         }
192 
193         /** @ignore */
194         function readFile(fileEntry){
195             if (fileEntry.isFile) {
196                 fileEntry.file(onFileSuccess, onFileError);
197             } else {
198                 throw new TypeError('Error: expecting file object');
199             }
200         }
201 
202         /**
203          * @public
204          * @lends uag.utils.makeFileExplorer
205          */
206         return {
207             /**
208              *  @param {object} $gridDiv JQuery Mobile object refering to a div with a 3-row grid class.
209              *  @param {function} fileCheckCb Function callback invoked when checking the chosen file. <em>Params: fileStr (string). Returns: boolean.</em>
210              *  @throws {TypeError} If $gridDiv type is not div[class="ui-grid-b"].
211              **/
212             run: function($gridDiv, fileCheckCb, closeCb) {
213                 // how to make sure this is called after onDeviceReady has fired?
214                 // TODO
215 
216                 // set params
217                 if ($gridDiv instanceof jQuery &&
218                     $gridDiv.is('div[class="ui-grid-b"]')) {
219                     gridDiv = $gridDiv;
220                 } else {
221                     throw new TypeError('Error: expecting explorer view objects');
222                 }
223                 onFileCheck = fileCheckCb;
224 
225                 // get root directory of the local file system
226                 if (gridDiv !== null) {
227                     if (rootDir !== null) {
228                         if (currentDir.name !== rootDir.name) {
229                             showDirectory(currentDir);
230                         } else {
231                             showDirectory(rootDir);
232                         }
233                     } else {
234                         window.requestFileSystem(LocalFileSystem.PERSISTENT, 0,
235                                                  onRequestFileSystemSuccess,
236                                                  onFileError);
237                     }
238                 } else {
239                     throw new Error('Error: explorer view objects are not set');
240                 }
241             },
242         };
243     };
244 
245     return parent;
246 }(uag || {}, jQuery, this, this.document));
