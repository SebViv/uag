204 	var onPageBeforeChange = function(event, data) {
205 		//alert("pagebeforechange");
206 		
207 		// TODO : upgrade this code specific 
208 		
209 		// We only want to handle changePage() calls where the caller is asking us to load a page by URL.
210 		if ( typeof data.toPage === "string" ) {
211 
212 			//alert(data.toPage);
213 			
214 			// We are being asked to load a page by URL, 
215 			// but we only want to handle URLs that request the data for a specific category. 
216 			var u = $.mobile.path.parseUrl( data.toPage );
217 			//var	re = '/^#' + containerPageSelector + '/';
218 			var	re = /^#container-page/;
219 
220 			//alert(re);
221 			//alert(u.hash);
222 			//alert(u.hash.search(re));
223 			
224 			if ( u.hash.search(re) !== -1 ) {
225 
226 				// We're being asked to display the items for a specific container.
227 				// Call our internal method that builds the content for the container on the fly based on our in-memory category data structure.
228 				//showCategory( u, data.options );
229 				
230 				containerId = u.hash.replace( /.*containerId=/, "" ); 
231 				var pageSelector = u.hash.replace( /\?.*$/, "" );
232 
233 				// alert(containerId + " - " + pageSelector + " - " + u.href);
234 				
235 				data.options.dataUrl = pageSelector;
236 				$.mobile.changePage( $(pageSelector), data.options);
237 				
238 				// Make sure to tell changePage() we've handled this call so it doesn't have to do anything.
239 				event.preventDefault();
240 			}
241 		}

244 	var onPageChange = function(event, data) {
245 		//alert("pagechange");
246 		var toPageId = data.toPage.attr("id");
247 		//alert(toPageId);
248 		switch(toPageId) {
249 		
250 			case containersListPageSelector : 
251 				//alert("containers list");
252 				renderContainersListPage();
253 				break;
254 				
255 			case containerPageSelector : 
256 				//alert("container");
257 				renderContainerPage();
258 				break;
259 				
260 			case setupPageSelector : 
261 				//alert("setup");
262 				renderSetupPage();
263 				break;
264 				
265 			default:
266 				break;
267 		}
268 	};

276 	var renderContainersListPageFooter = function() {
277 		//alert("renderContainersListTitle");
278 		var worker = dataContext.WorkerContainers.getWorker();
279 		 
280 		var footerText = "";
281 		if (worker.isAvailable) {
282 			var producer = dataContext.Producers.find();
283 			footerText = "( " + worker.name + " - " + producer.name + " )";
284 		} else {
285 			footerText = "( set worker and producer )";
286 		}
287 		$("#"+containersListPageFooterTextSelector).html(footerText);

290 	var renderContainersListPageContent = function() {
291 		//alert("renderContainersListContent");
292 		var containers = dataContext.WorkerContainers.getContainers();
293 		var view = $("#" + containersListPageContentListSelector);
294 		view.empty();
295 		
296 		if (containers.length > 0) {
297 			// products
298 			var products = dataContext.Producers.getProductsKeyNamePairs();
299 			// qualities
300 			var qualities = dataContext.Producers.getQualitiesKeyNamePairs();
301 			// markup generation 
302 			var containersMarkup = "";
303 			for (var i = 0 ; i < containers.length ; i++)
304 			{
305 				var container = containers[i];
306 				var date = new Date(container.ts); 
307 				containersMarkup = containersMarkup +  
308 					'<li>' + 
309 					'<a href="#' + containerPageSelector + '?containerId=' + container.id + '">' +
310 					'<span>' + 
311 					container.weight + ' kg of ' + 
312 					qualities[container.qualityId] + ' ' + 
313 					products[container.productId] + ' from ' + 
314 					date.toLocaleDateString() + 
315 					'</span>' + 
316 					'</a>' + 
317 					'</li>'; 
318 			}
319 			// ui update 
320 			view.html(containersMarkup);
321 		}
322 		// ui refresh 
323 		view.listview('refresh');
324 	};

326 	var renderContainerPage = function() {
327 		//alert("renderContainerPage");		
328 		var qualityId = null;
329 		var fieldId = null;
330 		var productId = null 
331 		
332 		if (undefined !== containerId && null !== containerId && containerId.length > 0) {
333 			// update container 
334 			var container = dataContext.WorkerContainers.findContainer(containerId);
335 			if (undefined !== container && null !== container) {
336 				qualityId = container.qualityId;
337 				fieldId = container.fieldId;
338 				productId = container.productId;
339 				$('#' + containerPageFormIdSelector).val(container.id);
340 				$("#" + containerPageFormWeightSelector).val(container.weight);
341 			}
342 		} else {
343 			// create container 
344 			$('#' + containerPageFormIdSelector).val("");
345 		}
346 		
347 		// render containers list page 
348 		renderQualitiesSelectList(qualityId);
349 		var fieldId = renderFieldsSelectList(fieldId);
350 		renderProductsSelectList(fieldId, productId);		
351 	};

389 	var renderSelectListOptions = function(selector, list, selectedValue) {
390 		// 
391 		var selectedItemId = "";
392 		var view = $("#" + selector);
393 		view.empty();
394 		// 
395 		if (undefined !== list && null !== list && list.length > 0) {
396 			// scope specific variables declarations
397 			var listOptionsMarkup = "";
398 			var selectedIndex = 0;
399 			// markup generation
400 			for (var i = 0 ; i < list.length ; i++) {
401 				listOptionsMarkup = listOptionsMarkup + 
402 					'<option value="' + list[i].id + '">' + list[i].name + '</option>';
403 				if (list[i].id === selectedValue) {
404 					selectedIndex = i;
405 				}
406 			}
407 			view.html(listOptionsMarkup);
408 			// this old school javascript does work 
409 			document.getElementById(selector).selectedIndex = selectedIndex;
410 			// but this jQuery based solution doesn't work
411 			//view.selectedIndex = selectedIndex;
412 			// TODO : so search for jQuery solution
413 			
414 			selectedItemId = list[selectedIndex].id;
415 		}
416 		view.selectmenu("refresh");
417 		return selectedItemId;
418 	};

433 	var onSaveWorkerTapped = function() {
434 		//alert("onSaveWorkerTapped");
435 		var workerNameTextInput = $("#" + setupPageFormNameSelector);
436 		var producersSelectList = $("#" + setupPageFormProducerSelector);
437 		var uuid = device.uuid;
438 		// set new worker 
439 		dataContext.WorkerContainers.setWorker(uuid, workerNameTextInput.val(), producersSelectList.val());

441 		$.mobile.changePage($("#" + containersListPageSelector));

444 	var onSaveContainerTapped = function() {
445 		//alert("onSaveContainerTapped");
446 		// get user input 
447 		var idHiddenInput = $("#" + containerPageFormIdSelector);
448 		var weightTextInput = $("#" + containerPageFormWeightSelector);
449 		var qualitiesSelectList = $("#" + containerPageFormQualitiesSelector);
450 		var fieldsSelectList = $("#" + containerPageFormFieldsSelector);
451 		var productsSelectList = $("#" + containerPageFormProductsSelector);
452 
453 		var id = idHiddenInput.val();
454 		var weight = weightTextInput.val();
455 		var qualityId = qualitiesSelectList.val();
456 		var fieldId = fieldsSelectList.val();
457 		var productId = productsSelectList.val();
458 		
459 		//alert(id + " - " + weight + " - " + qualityId + " - " + fieldId + " - " + productId);
460 		
461 		// validate user input
462 		var valid = true;
463 		
464 		// do task 
465 		if (valid) {
466 			if (undefined !== id && null !== id && id.length > 0) {
467 				// update container
468 				dataContext.WorkerContainers.updateContainer(id, weight,qualityId,fieldId,productId);
469 			} else {
470 				// add container 
471 				dataContext.WorkerContainers.addContainer(weight,qualityId,fieldId,productId);
472 			}
473 		} else {
474 			// TODO : error message 
475 		}
476 		
477 		// redirect to containers list page 
478 		$.mobile.changePage($("#" + containersListPageSelector));
479 	};
