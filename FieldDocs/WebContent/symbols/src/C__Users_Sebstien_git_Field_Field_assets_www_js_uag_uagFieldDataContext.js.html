121 		if (worker.isAvailable) {
122 			// get producer's list 
123 			var producers = Producers.get();
124 			// filter producers 
125 			for ( var int = 0; int < producers.length; int++) {
126 				// if ids match, 
127 				//alert(worker.producerId + " - " + producers[int].id);
128 				if (worker.producerId === producers[int].id) {
129 					// then return it
130 					return producers[int];
131 				}
132 			}
133 		}
134 		return null;
135 	};

586 	 * @name uag.field.dataContext.Import.addProducer
587 	 * @function
588 	 * @static
589 	 * @param jsonProducerData producer json data 
590 	 * @throws {Error} Producer already imported
591 	 * @returns {String} producer's id
592 	 */
593 	Import.addProducer = function(jsonProducerData) {
594 		
595 		// get new producer's id 
596 		var producerId = jsonProducerData.id;
597 		// 
598 		var exist = false;
599 		// producers 
600 		var producers = Producers.get();
601 		// search
602 		for ( var int = 0 ; int < producers.length ; int++ ) {
603 			exist = exist || producers[int].id === producerId;
604 		}
605 
606 		// if not exists already
607 		if (!exist) {
608 			// then add it to producers list
609 			
610 			// add json producer data to json producers list 
611 			producers.push(jsonProducerData);
612 			// set json producers list to local storage   
613 			StorageSave(Producers.getLocalStorageKey(), producers);
614 			
615 			// return producer's id
616 			return jsonProducerData.id;
617 			
618 		} else {
619 			throw new Error("Producer already imported");
620 		}
621 		
622 	};
